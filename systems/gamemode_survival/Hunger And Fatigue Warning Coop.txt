:H_WARN2
thread 'H_WARN2'             

create_thread @H_DSP2  
wait 2000

:WARNING_H_CHECK_COOP
wait 0
if
not $SATIET2_MEDINUM_WARNING == 1
jf @HUNGRY_MEDINUM_WARNING_COOP
if
not $SATIET2_LOW_WARNING == 1
jf @HUNGRY_LOW_WARNING_COOP
jump @WARNING_H_CHECK_COOP

:HUNGRY_MEDINUM_WARNING_COOP
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS_2P == 0
AFTEREAT_TIMER_2P == 0  
jf @WARNING_H_CHECK2_COOP
03E5: show_text_box 'INV10B'
wait 10000
jump @WARNING_H_CHECK2_COOP 

:HUNGRY_LOW_WARNING_COOP
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS_2P == 0
AFTEREAT_TIMER_2P == 0  
jf @WARNING_H_CHECK3_COOP
03E5: show_text_box 'INV11B'
wait 10000
jump @WARNING_H_CHECK3_COOP

:WARNING_H_CHECK2_COOP
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS_2P == 0
AFTEREAT_TIMER_2P == 0  
jf @WARNING_H_CHECK_COOP
wait 0
if 
$SATIET2_MEDINUM_WARNING == 0 
jf @WARNING_H_CHECK2_COOP
03D5: remove_text 'INV10B'
jump @WARNING_H_CHECK_COOP

:WARNING_H_CHECK3_COOP
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS_2P == 0
AFTEREAT_TIMER_2P == 0
jf @WARNING_H_CHECK_COOP
wait 0
if 
$SATIET2_LOW_WARNING == 0
jf @WARNING_H_CHECK3_COOP  
03D5: remove_text 'INV11B'
jump @WARNING_H_CHECK_COOP

:F_WARN2
thread 'F_WARN2'
                
create_thread @F_DSP2  
wait 2000

:WARNING_COOP_B_CHECK
wait 0
if
not $FATIGU2_MEDINUM_WARNING == 1
jf @BORING_MEDINUM_WARNING_COOP
if
not $FATIGU2_HIGH_WARNING == 1
jf @BORING_HIGH_WARNING_COOP
jump @WARNING_COOP_B_CHECK

:BORING_MEDINUM_WARNING_COOP 
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
jf @WARNING_COOP_B_CHECK
03E5: show_text_box 'INV61B'
wait 2000
jump @WARNING_COOP_B_CHECK2

:BORING_HIGH_WARNING_COOP
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
jf @WARNING_COOP_B_CHECK
03E5: show_text_box 'INV62B'
wait 2000
jump @WARNING_COOP_B_CHECK3

:WARNING_COOP_B_CHECK2
wait 0
if 
$FATIGU2_MEDINUM_WARNING == 0  
jf @WARNING_COOP_B_CHECK2
03D5: remove_text 'INV61B'
jump @WARNING_COOP_B_CHECK

:WARNING_COOP_B_CHECK3
wait 0
if 
$FATIGU2_HIGH_WARNING == 0
jf @WARNING_COOP_B_CHECK3
03D5: remove_text 'INV62B'
jump @WARNING_COOP_B_CHECK

:H_DSP2
thread 'H_DSP2'                 

:HUNGRY_COOP_NORMAL
wait 0

while 8800: not  in_two_players_mode
wait 0
end
 
if  
$SURVIVOR_SATIETY[COOP_ACTOR_SURV_NUM] > 149.9
jf @HUNGRY_COOP_MEDINUM
$SATIET2_MEDINUM_WARNING = 0
$SATIET2_LOW_WARNING = 0

:HUNGRY_COOP_MEDINUM
if and
$SURVIVOR_SATIETY[COOP_ACTOR_SURV_NUM] > 49.9 
$SURVIVOR_SATIETY[COOP_ACTOR_SURV_NUM] < 149.9
jf @HUNGRY_COOP_LOW
if and
$EATING_STATUS_2P == 0 
$SAVING_DELAY == 0  
jf @HUNGRY_COOP_MEDINUM2
$SATIET2_MEDINUM_WARNING = 1
:HUNGRY_COOP_MEDINUM2
$SATIET2_LOW_WARNING = 0

:HUNGRY_COOP_LOW
wait 0

while 8800: not  in_two_players_mode
wait 0
end

if 
$SURVIVOR_SATIETY[COOP_ACTOR_SURV_NUM] < 49.9 
jf @HUNGRY_COOP_NORMAL
if and
$EATING_STATUS_2P == 0 
$SAVING_DELAY == 0  
jf @HUNGRY_COOP_LOW
$SATIET2_MEDINUM_WARNING = 0
$SATIET2_LOW_WARNING = 1  
jump @HUNGRY_COOP_NORMAL

:F_DSP2
thread 'F_DSP2'                

:BORING_SMALL_COOP
wait 0 
if                                    
$FATIGUE_COOP_LEVEL < 49.9  
jf @BORING_NORMAL_COOP
$FATIGU2_MEDINUM_WARNING = 0
$FATIGU2_HIGH_WARNING = 0

:BORING_NORMAL_COOP
if and
$FATIGUE_COOP_LEVEL > 49.9 
$FATIGUE_COOP_LEVEL < 79.9
jf @BORING_HIGH_COOP
$FATIGU2_MEDINUM_WARNING = 1
$FATIGU2_HIGH_WARNING = 0

:BORING_HIGH_COOP
if
$FATIGUE_COOP_LEVEL > 79.9
jf @BORING_SMALL_COOP
$FATIGU2_MEDINUM_WARNING = 0
$FATIGU2_HIGH_WARNING = 1
jump @BORING_SMALL_COOP
