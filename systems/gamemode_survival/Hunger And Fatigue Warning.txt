:H_WARN
thread 'H_WARN'        
        
create_thread @H_DSP  
wait 2000      

:WARNING_H_CHECK
wait 0
if
not $SATIETY_MEDINUM_WARNING == 1
jf @HUNGRY_MEDINUM_WARNING
if
not $SATIETY_LOW_WARNING == 1
jf @HUNGRY_LOW_WARNING
jump @WARNING_H_CHECK

:HUNGRY_MEDINUM_WARNING
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS == 0
AFTEREAT_TIMER_1P == 0  
jf @WARNING_H_CHECK2
03E5: show_text_box 'INV10'
wait 10000
jump @WARNING_H_CHECK2 

:HUNGRY_LOW_WARNING
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS == 0
AFTEREAT_TIMER_1P == 0  
jf @WARNING_H_CHECK3
03E5: show_text_box 'INV11'
wait 10000
jump @WARNING_H_CHECK3

:WARNING_H_CHECK2
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS == 0
AFTEREAT_TIMER_1P == 0  
jf @WARNING_H_CHECK
wait 0
if 
$SATIETY_MEDINUM_WARNING == 0 
jf @WARNING_H_CHECK2
03D5: remove_text 'INV10'
jump @WARNING_H_CHECK

:WARNING_H_CHECK3
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
$EATING_STATUS == 0
AFTEREAT_TIMER_1P == 0
jf @WARNING_H_CHECK
wait 0
if 
$SATIETY_LOW_WARNING == 0
jf @WARNING_H_CHECK3  
03D5: remove_text 'INV11'
jump @WARNING_H_CHECK

:F_WARN
thread 'F_WARN'
                
create_thread @F_DSP  
wait 2000

:WARNING_B_CHECK
wait 0
if
not $FATIGUE_MEDINUM_WARNING == 1
jf @BORING_MEDINUM_WARNING
if
not $FATIGUE_HIGH_WARNING == 1
jf @BORING_HIGH_WARNING
jump @WARNING_B_CHECK

:BORING_MEDINUM_WARNING 
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
jf @WARNING_B_CHECK
03E5: show_text_box 'INV61'
wait 2000
jump @WARNING_B_CHECK2

:BORING_HIGH_WARNING
if and
$HUD_REMOVE == 0
$SAVING_DELAY == 0
jf @WARNING_B_CHECK
03E5: show_text_box 'INV62'
wait 2000
jump @WARNING_B_CHECK3

:WARNING_B_CHECK2
wait 0
if 
$FATIGUE_MEDINUM_WARNING == 0  
jf @WARNING_B_CHECK2
03D5: remove_text 'INV61'
jump @WARNING_B_CHECK

:WARNING_B_CHECK3
wait 0
if 
$FATIGUE_HIGH_WARNING == 0
jf @WARNING_B_CHECK3
03D5: remove_text 'INV62'
jump @WARNING_B_CHECK

:H_DSP
thread 'H_DSP'                 

:HUNGRY_NORMAL
wait 0 
if  
$SATIETY_LEVEL > 149.9
jf @HUNGRY_MEDINUM
$SATIETY_MEDINUM_WARNING = 0
$SATIETY_LOW_WARNING = 0

:HUNGRY_MEDINUM
if and
$SATIETY_LEVEL > 49.9 
$SATIETY_LEVEL < 149.9
jf @HUNGRY_LOW
if
$SAVING_DELAY == 0  
jf @HUNGRY_MEDINUM2
$SATIETY_MEDINUM_WARNING = 1
:HUNGRY_MEDINUM2
$SATIETY_LOW_WARNING = 0

:HUNGRY_LOW
wait 0
if 
$SATIETY_LEVEL < 49.9 
jf @HUNGRY_NORMAL
if
$SAVING_DELAY == 0  
jf @HUNGRY_LOW
$SATIETY_MEDINUM_WARNING = 0
$SATIETY_LOW_WARNING = 1  
jump @HUNGRY_NORMAL

:F_DSP
thread 'F_DSP'                

:BORING_SMALL
wait 0 
if                                    
$FATIGUE_LEVEL < 49.9  
jf @BORING_NORMAL
$FATIGUE_MEDINUM_WARNING = 0
$FATIGUE_HIGH_WARNING = 0

:BORING_NORMAL
if and
$FATIGUE_LEVEL > 49.9 
$FATIGUE_LEVEL < 79.9
jf @BORING_HIGH
$FATIGUE_MEDINUM_WARNING = 1
$FATIGUE_HIGH_WARNING = 0

:BORING_HIGH
if
$FATIGUE_LEVEL > 79.9
jf @BORING_SMALL
$FATIGUE_MEDINUM_WARNING = 0
$FATIGUE_HIGH_WARNING = 1
jump @BORING_SMALL
